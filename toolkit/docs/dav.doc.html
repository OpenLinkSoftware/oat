Displays a WebDav Browser window for file/folder picking/opening/saving.

Functions

  • OAT.WebDav.getFileName(user, password, path, onlyExistingFiles, buttonLabel,callback)
  • OAT.WebDav.getFile(user, password, path, onlyExistingFiles, buttonLabel,callback)
  • OAT.WebDav.saveFile(user,password,path, fileContent, displayUI, callback)

OAT.WebDav.init(options)

Set values for main configuration optons .

OAT.WebDav.open(options)

Set values for current configuration optons(this action).

Argument options has the following properties:

  • container - name of WebDav Browser window (default:my_browser)
  • mode - type of open windows (default:browser)
    • browser - choose file from file list - return path and file name
    • open_dialog - choose file from file list - return path,file name and file content
    • save_dialog - choose file from file list - send new content of file
  • toolbar - set which actions to be visible in toolbar
    • new_folder - option to create new folder (default:true)
    • change_view - option to change list style (default:true)
  • user - Dav user name (default:'')
  • pass - Dav user password (default:'')
  • pathDefault - Path that must be open as default if browser is open for first time (default:'/')
  • filename - file name
  • file_ext - Default file extension. Will be added if chosen file name doesn't (default:'')
  • width - Width size in pixels of DavBrowser Window (default:750)
  • height - Width size in pixels of DavBrowser Window (default:420)
  • x - X-position in pixels of DavBrowser Window (default:120)
  • y - Y-position in pixels of DavBrowser Window (default:120)
  • imagePath - Path to icons(default:'/DAV/JS/images/')
  • imageExt - File extension for icons (default:'gif')
  • onConfirmClick - callback function which executes after OK Button is clicked. Params are chosen path,filename,content(if available)
  • afterSave- callback function which executes after saving of file. Params are chosen path,filename
  • dontDisplayWarning - suspend confirm dialog on file replacing (default:false)

var options = {
imagePath:'../images/',
imageExt:'gif',
pathDefault:'/DAV/home/demo/Public/',
};
OAT.WebDav.init(options);

var dav_browse = function(){
var options = {
mode:'browser',
onOKClick:function(path,fname){
document.f1.my_dav_file.value = path + fname;

}
};
OAT.WebDav.open(options);
}

var dav_open = function(){
options = {
mode:'open_dialog',
onOpenClick:function(path,fname,data){
document.f1.my_dav_content.value = data;
$('my_dav_path').innerHTML = path;
$('my_dav_filename').innerHTML = fname;
return true;
}
};
OAT.WebDav.open(options);
}

var dav_save = function(){
var options = {
mode:'save_dialog',
dontDisplayWarning:true,
onSaveClick:function() { return document.f1.my_dav_content.value;},
afterSave:function(path,fname) {
$('my_dav_path').innerHTML = path;
$('my_dav_filename').innerHTML = fname;
}
};
OAT.WebDav.open(options);
}
